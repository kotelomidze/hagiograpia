        function renderSidebar() {
            navList.innerHTML = '';
            
            function createItem(node, depth = 0) {
                const isFolder = node.type === 'folder';
                const paddingLeft = (depth * 1.5 + 0.75) + 'rem';
                
                const div = document.createElement('div');
                div.className = 'select-none';

                const header = document.createElement('div');
                header.className = 'nav-item flex items-center py-2 px-3 cursor-pointer transition-colors duration-200 text-stone-700';
                header.style.paddingLeft = paddingLeft;
                
                if (activeNode && activeNode.id === node.id) {
                    header.classList.add('active');
                }

                const iconSpan = document.createElement('span');
                iconSpan.className = 'mr-2 opacity-70';
                if (isFolder) {
                    iconSpan.innerHTML = ``; 
                } else {
                    iconSpan.innerHTML = ``; 
                }
                header.appendChild(iconSpan);

                const titleSpan = document.createElement('span');
                titleSpan.className = 'truncate';
                titleSpan.textContent = node.title;
                header.appendChild(titleSpan);

                div.appendChild(header);

                header.addEventListener('click', () => {
                    if (isFolder) {
                        return; 
                    }
                    selectNode(node);
                    if (window.innerWidth < 768) {
                        toggleSidebar(false);
                    }
                });

                if (isFolder && node.children) {
                    const childrenContainer = document.createElement('div');
                    iconSpan.innerHTML = ``;
                    
                    node.children.forEach(child => {
                        childrenContainer.appendChild(createItem(child, depth + 1));
                    });
                    div.appendChild(childrenContainer);
                }

                return div;
            }

            SITE_DATA.forEach(node => {
                navList.appendChild(createItem(node));
            });
        }

        function selectNode(node) {
            activeNode = node;
            
            emptyState.classList.add('hidden');
            toolbar.classList.remove('hidden');
            
            filenameDisplay.textContent = node.fileName;
            
            previewView.innerHTML = node.content;
            
            const fullCode = generateFullHtml(node);
            codeContent.textContent = fullCode;
            
            setMode('preview');
            
            renderSidebar();
        }

        function setMode(mode) {
            viewMode = mode;
            if (mode === 'preview') {
                previewView.classList.remove('hidden');
                codeView.classList.add('hidden');
                
                btnPreview.classList.add('bg-amber-100', 'text-amber-800', 'font-medium');
                btnPreview.classList.remove('text-stone-600', 'hover:bg-stone-100');
                
                btnCode.classList.remove('bg-amber-100', 'text-amber-800', 'font-medium');
                btnCode.classList.add('text-stone-600', 'hover:bg-stone-100');
            } else {
                previewView.classList.add('hidden');
                codeView.classList.remove('hidden');
                
                btnCode.classList.add('bg-amber-100', 'text-amber-800', 'font-medium');
                btnCode.classList.remove('text-stone-600', 'hover:bg-stone-100');
                
                btnPreview.classList.remove('bg-amber-100', 'text-amber-800', 'font-medium');
                btnPreview.classList.add('text-stone-600', 'hover:bg-stone-100');
            }
        }

        function downloadCurrent() {
            if (!activeNode) return;
            const fullCode = generateFullHtml(activeNode);
            const element = document.createElement("a");
            const file = new Blob([fullCode], {type: 'text/html'});
            element.href = URL.createObjectURL(file);
            element.download = activeNode.fileName;
            document.body.appendChild(element); 
            element.click();
            document.body.removeChild(element);
        }

        function toggleSidebar(show) {
            if (show) {
                sidebar.classList.remove('-translate-x-full');
                sidebarOverlay.classList.remove('hidden');
            } else {
                sidebar.classList.add('-translate-x-full');
                sidebarOverlay.classList.add('hidden');
            }
        }

        btnPreview.addEventListener('click', () => setMode('preview'));
        btnCode.addEventListener('click', () => setMode('code'));
        btnDownload.addEventListener('click', downloadCurrent);
        
        openSidebarBtn.addEventListener('click', () => toggleSidebar(true));
        closeSidebarBtn.addEventListener('click', () => toggleSidebar(false));
        sidebarOverlay.addEventListener('click', () => toggleSidebar(false));

        renderSidebar();
